						import turtle
						import numpy
						import math

						def Read_OBJ():
						data = open("Untitled.obj","r")
						line = data.readline()

						while line:
							record = line.split()
							if "v" in record:
								record.remove("v")
								vertices.append(record)

							if "f" in record:
								record.remove("f")
								segment = " ".join(record)
								record = segment.split()

								for i in record:
									if i != "":
										string = ""
										for x in i:
											if x != "/":
												string += x
											else:
												break
										faces.append(string)

								faces.append("end")
								
							line  = data.readline()
						Draw_OBJ(0,0)
							
						def Draw_OBJ(x,y):
						bob = turtle.Turtle()
						screen = turtle.Screen()
						screen.screensize(1920, 1920)

						turtle.delay(0)
						turtle.tracer(0, 0)

						bob.speed = (0)
						bob.ht()
						bob.penup()
						bob.setpos((float(vertices[int(faces[0]) - 1][0]) + x) * 100, 
									(float(vertices[int(faces[0]) - 1][1]) + y) * 100)
						bob.pendown()

						start_seg = 0

						cur_steps = 0
						steps = -1

						for i in range(len(faces)):
							if faces[i] == "end":
								bob.setpos(
									(float(vertices[int(faces[start_seg]) - 1][0]) + x) * 100,
									(float(vertices[int(faces[start_seg]) - 1][1]) + y) * 100)
									
								bob.penup()
								start_seg = i + 1

								if cur_steps == steps:
									turtle.update()
									cur_steps = 0
									
								if steps > 0:
									cur_steps += 1
								continue
								
							bob.setpos((float(vertices[int(faces[i]) - 1][0]) + x) * 100, 
										(float(vertices[int(faces[i]) - 1][1]) + y) * 100)
							bob.pendown()

						vertices = []
						faces = []
						Read_OBJ()
